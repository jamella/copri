This is a test of copri two_power function.

#include <stdlib.h>
#include <stdio.h>
#include <gmp.h>
#include "test.h"
#include "copri.h"

int tests_passed = 0;
int tests_failed = 0;

Compute a^2^n and compare it to expect.

static char * test(char * a, unsigned long long n, char * expect) {
	mpz_t v;
	mpz_t e;

Initialize v and e.

	mpz_init_set_str(v, a, 10);
	mpz_init_set_str(e, expect, 10);

Execute the two_power function.

	two_power(v, n);

Check the return value.

	if (mpz_cmp(v, e) != 0) {
		return mpz_get_str(NULL, 10, v);
	}

Free memory.

	mpz_clear(v);
	mpz_clear(e);
	return 0;
}

Run all tests.
